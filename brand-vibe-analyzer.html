    // Initialize quiz

    function initQuiz() {

        displayQuestion();

        updateProgress();

    // Display current question

    function displayQuestion() {

        const question = quizData[currentQuestion];

        document.getElementById('questionText').textContent = question.question;

        document.getElementById('questionCounter').textContent = `${currentQuestion + 1} of ${quizData.length}`;

        

        const container = document.getElementById('answersContainer');

        container.innerHTML = '';

        

        question.options.forEach((option, index) => {

            const label = document.createElement('label');

            const borderClass = option.border ? 'border' : '';

            label.className = 'flex items-center p-4 border-2 border-cream-beige rounded-xl hover:border-spotlight-yellow cursor-pointer transition-all answer-option group';

            label.innerHTML = `

                <input type="radio" name="answer" value="${index}" class="mr-4 text-spotlight-yellow scale-125">

                    <div class="w-8 h-8 bg-gradient-to-r ${option.color} rounded-lg ${borderClass}"></div>

                    <span class="group-hover:text-jet-black transition-colors">${option.text}</span>

                </div>

            `;

            container.appendChild(label);

        });


        // Update button states

        document.getElementById('prevBtn').disabled = currentQuestion === 0;

        document.getElementById('nextBtn').textContent = currentQuestion === quizData.length - 1 ? 'Reveal My Brand Vibe ✨' : 'Next Question ✨';

    }


    // Update progress bar

    function updateProgress() {

        const progress = ((currentQuestion + 1) / quizData.length) * 100;

        document.getElementById('progressBar').style.width = `${progress}%`;

    }


    // Handle next button click

    document.getElementById('nextBtn').addEventListener('click', function() {

        const selectedAnswer = document.querySelector('input[name="answer"]:checked');

        

        if (!selectedAnswer) {

            alert('Please select an answer before continuing.');

            return;

        }


        // Store answer and update style scores

        const answerIndex = parseInt(selectedAnswer.value);

        userAnswers[currentQuestion] = answerIndex;

        styleScores[answerIndex]++;

        

        // Update style preferences in real-time

        updateStylePreferences();


        if (currentQuestion < quizData.length - 1) {

            currentQuestion++;

            displayQuestion();

            updateProgress();

            updateFeedback();

        } else {

            showResults();

        }

    });


    // Handle previous button click

    document.getElementById('prevBtn').addEventListener('click', function() {

        if (currentQuestion > 0) {

            // Remove previous answer from scores

            if (userAnswers[currentQuestion] !== undefined) {

                styleScores[userAnswers[currentQuestion]]--;

            }

            

            currentQuestion--;

            displayQuestion();

            updateProgress();

            

            // Restore previous answer if exists

            if (userAnswers[currentQuestion] !== undefined) {

                const radios = document.querySelectorAll('input[name="answer"]');

                radios[userAnswers[currentQuestion]].checked = true;

            }

            

            updateStylePreferences();

        }

    });


    // Update style preferences display

    function updateStylePreferences() {

        const total = styleScores.reduce((a, b) => a + b, 0);

        const styleElements = document.querySelectorAll('[class*="bg-gradient-to-r"]');

        const percentageElements = document.querySelectorAll('.text-xs.font-medium.text-gray-600');

        

        styleScores.forEach((score, index) => {

            const percentage = total > 0 ? Math.round((score / total) * 100) : 0;

            if (styleElements[index + 5]) { // Skip the first 5 elements (they're in the quiz)

                styleElements[index + 5].style.width = `${percentage}%`;

            }

            if (percentageElements[index]) {

                percentageElements[index].textContent = `${percentage}%`;

            }

        });

    }

        }

    }


    // Show results modal

    function showResults() {

        // Find the dominant brand vibe

        const maxScore = Math.max(...styleScores);

        const dominantVibeIndex = styleScores.indexOf(maxScore);

        const dominantVibe = brandVibes[dominantVibeIndex];

        

        // Calculate confidence based on how dominant the top choice is

        const total = styleScores.reduce((a, b) => a + b, 0);

        const confidence = Math.round((maxScore / total) * 100);

        

        // Update modal content

        document.getElementById('resultIcon').innerHTML = `<span class="text-2xl">${dominantVibe.icon}</span>`;

        document.getElementById('resultIcon').style.background = dominantVibe.background;

        document.getElementById('brandVibeTitle').textContent = dominantVibe.name;

        document.getElementById('brandVibeDescription').textContent = dominantVibe.description;

        document.getElementById('finalScore').textContent = `${quizData.length}/${quizData.length}`;

        document.getElementById('finalConfidence').textContent = `${confidence}%`;

        

        document.getElementById('resultsModal').classList.remove('hidden');

        document.getElementById('resultsModal').classList.add('flex');

    }


    // Close results modal

    function closeResults() {

        document.getElementById('resultsModal').classList.add('hidden');

        document.getElementById('resultsModal').classList.remove('flex');

    }


    // Restart quiz

    function restartQuiz() {

        currentQuestion = 0;

        userAnswers = [];

        styleScores = [0, 0, 0, 0, 0];

        closeResults();

        initQuiz();

        updateStylePreferences();

    }

                option.classList.remove('border-spotlight-yellow', 'bg-spotlight-yellow/5');

    // Add sparkle cursor effect

        if (Math.random() < 0.1) { // 10% chance to create sparkle

            createSparkle(e.clientX, e.clientY);

        }

    });


    function createSparkle(x, y) {

        const sparkle = document.createElement('div');

        sparkle.innerHTML = '✨';

        sparkle.style.position = 'fixed';

        sparkle.style.left = x + 'px';

        sparkle.style.top = y + 'px';

        sparkle.style.pointerEvents = 'none';

        sparkle.style.zIndex = '9999';

        sparkle.style.fontSize = '12px';

        sparkle.style.animation = 'sparkleFloat 1s ease-out forwards';

        

        document.body.appendChild(sparkle);

        

        setTimeout(() => {

            sparkle.remove();

        }, 1000);

    }


    // Add sparkle animation CSS

    const style = document.createElement('style');

    style.textContent = `

        @keyframes sparkleFloat {

            0% {

                opacity: 1;

                transform: translateY(0px) scale(1);

            }

            100% {

                opacity: 0;

                transform: translateY(-20px) scale(0.5);

            }

        }

    `;

    document.head.appendChild(style);
